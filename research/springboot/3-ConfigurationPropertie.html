# 3.生成spring-configuration-metadata.json文件

a. 添加maven插件

 <build>
        <plugins>
            <plugin>
                <groupId>org.apache.maven.plugins</groupId>
                <artifactId>maven-compiler-plugin</artifactId>
                <version>3.8.0</version>
                <configuration>
                    <annotationProcessorPaths>
                        <annotationProcessorPath>
                            <groupId>org.springframework.boot</groupId>
                            <artifactId>spring-boot-configuration-processor</artifactId>
                            <version>2.1.5.RELEASE</version>
                        </annotationProcessorPath>
                    </annotationProcessorPaths>
                </configuration>
            </plugin>
        </plugins>
    </build>

b. maven clean

c. maven compile

d. 删除maven插件

e. 将target/classes/META-INFO/spring-configuration-metadata.json，拷贝到src/resources/META-INF中

# 附录：ConditionalOnProperty

使用常见于根据用户的配置参数决定是否加载相关配置类

# 使用

@Configuration
@ConditionalOnProperty(prefix = "mvc.enable", name = "enable", havingValue = "true", matchIfMissing = false)
public class RestMvcAutoConfiguration {

    ...

# 参数

  • prefix: 参数前缀
  • name: 参数名
  • havingValue: 需要的参数值
  • matchIfMissing: 缺失是否算匹配
